import json
import re
from datetime import datetime
import time #   
import matplotlib.pyplot as plt
from google import genai
from google.genai.errors import APIError
from google.genai import types


GEMINI_API_KEY = "     -"  

SERVICE_PROMPT = "      .           ,         .       , , .                  .   ."
#  < 0 ();  > 0 ()
SENTIMENT_DICT = {
    '': -2, '': -1.5, '': -0.5, '': -1, '': -1.5, '': -2,
    '': -1, '': -0.5, '': -0.3,
    '': 1.5, '': 2, '': 3, '': 2.5, '': 2, '': 1.5,
    '': 1, '': 1.5, '': 2.5, '': 1, '': 3,
}
#                  

# ----------------------------------------------------

class ServiceManager:
    """
    """
    def __init__(self, api_key: str, system_prompt: str, sentiment_dict: dict):
        """
          .
        """
        self.api_key = api_key
        self.system_prompt = system_prompt
        self.sentiment_dict = sentiment_dict
        self.chat_history = [] 
        try:
            self.client = genai.Client(api_key=self.api_key)
            
            #   config   -system_instruction 
            config = types.GenerateContentConfig(
                system_instruction=self.system_prompt
            )

            
            self.chat = self.client.chats.create(
                model="gemini-2.5-flash", 
                config=config 
            )
            
            print("      .  ' ' .")
        except Exception as e:
            print(f"  API/:    -API     genai. : {e}")
            self.client = None
            self.chat = None
    def _normalize_text(self, text: str) -> str:
        """  :      ."""
        #    
        text = re.sub(r'[.,!?;:()"]', '', text)
        return text.lower().strip()

    def calculate_sentiment(self, text: str) -> float:
        """
               .
        
        """
        normalized_text = self._normalize_text(text)
        words = normalized_text.split()
        
        #     List Comprehension
        scores = [self.sentiment_dict.get(word, 0) for word in words]
        
        return sum(scores)
        
    def _add_message_to_history(self, sender: str, text: str):
        """       ."""
        score = self.calculate_sentiment(text)
        message = {
            'sender': sender,
            'text': text,
            'sentiment_score': score,
            'timestamp': time.time()
        }
        self.chat_history.append(message)

    def communicate_with_api(self, user_input: str) -> str:
        """
          API Gemini,     (  -Chat), 
         .
        """
        if not self.chat:
            return "      ."

        # 1.     
        self._add_message_to_history(sender='User', text=user_input)
        
        try:
            #  -API       
            response = self.chat.send_message(user_input)
            
            bot_response = response.text
            
            # 2.     
            self._add_message_to_history(sender='Bot', text=bot_response)
            
            return bot_response
            
        except APIError as e:
            #     -API
            self.chat_history.pop() #      
            print(f"  API  : {e}")
            return "   .   ."
        except Exception as e:
            self.chat_history.pop()
            print(f"   : {e}")
            return "   .   ."


    # ---  /   ---

    def save_to_json(self, filename="chat_log.json"):
        """      JSON,    ."""
        try:
            with open(filename, 'w', encoding='utf-8') as f:
                json.dump(self.chat_history, f, ensure_ascii=False, indent=4)
            print(f"\n    : **{filename}**")
            return filename
        except IOError as e:
            #    ()
            print(f"   : {e}")
            return None

    def load_from_json(self, filename: str):
        """    JSON ,    ."""
        try:
            with open(filename, 'r', encoding='utf-8') as f:
                self.chat_history = json.load(f)
            print(f"     : {filename}")
        except FileNotFoundError:
            print(f"  :  '{filename}'  .")
        except json.JSONDecodeError:
            print(f"  :  '{filename}'   JSON .")
        except IOError as e:
            #    ()
            print(f"   : {e}")
            
    # ---     (Analytics) ---

    def _get_sentiment_iterator(self, sender_filter: str = None):
        """
        Generator -  advance python.
          (Generator)      .
        """
        for message in self.chat_history:
            if sender_filter is None or message['sender'] == sender_filter:
                yield message['sentiment_score']
                
    def analyze_and_plot_results(self, filename: str):
        """  ,      ."""
        self.load_from_json(filename)

        if not self.chat_history:
            print("     .")
            return
        font_name = 'Miriam' 
        
        # 2.         
        plt.rcParams['font.sans-serif'] = [font_name]
        plt.rcParams['font.family'] = 'sans-serif'
        
        # 3.  -matplotlib    ( )    
        plt.rcParams['axes.unicode_minus'] = False

        # 1.    
        user_messages = [msg for msg in self.chat_history if msg['sender'] == 'User']
        bot_messages = [msg for msg in self.chat_history if msg['sender'] == 'Bot']
        
        # 2.    
        user_scores = list(self._get_sentiment_iterator(sender_filter='User'))
        bot_scores = list(self._get_sentiment_iterator(sender_filter='Bot'))
        
        avg_user_sentiment = sum(user_scores) / len(user_scores) if user_scores else 0
        avg_bot_sentiment = sum(bot_scores) / len(bot_scores) if bot_scores else 0
        
        #   (User Sentiment)
        cumulative_sentiment = [sum(user_scores[:i+1]) for i in range(len(user_scores))]

        # 3.    
        
        #     ( > 0)  Dictionary Comprehension
        negative_words_dict = {word: score for word, score in self.sentiment_dict.items() if score > 0}
        
        negative_word_count = {}
        for message in user_messages:
            text = self._normalize_text(message['text'])
            for word in text.split():
                if word in negative_words_dict:
                    negative_word_count[word] = negative_word_count.get(word, 0) + 1

        # 4.  
        plt.style.use('ggplot')
        
        # ---  1:   (Line Chart) ---
        plt.figure(figsize=(10, 6))
        plt.plot(range(1, len(user_scores) + 1), cumulative_sentiment, marker='o', linestyle='-', color='darkblue')
        plt.axhline(0, color='gray', linestyle='--') #  
        plt.title('  :      ')
        plt.xlabel('  ( )')
        plt.ylabel('   ( < 0,  > 0)')
        plt.grid(True)
        plt.show()

        # ---  2:   (Bar Chart) ---
        plt.figure(figsize=(8, 5))
        senders = [' (User)', ' (Bot)']
        averages = [avg_user_sentiment, avg_bot_sentiment]
        colors = ['red' if avg > 0 else 'green' for avg in averages]
        
        plt.bar(senders, averages, color=colors)
        plt.axhline(0, color='black', linestyle='-')
        plt.title('  :   -   ')
        plt.ylabel('   ( < 0,  > 0)')
        plt.show()

        # ---  3:    (Bar Chart) ---
        if negative_word_count:
            plt.figure(figsize=(10, 6))
            words = list(negative_word_count.keys())
            counts = list(negative_word_count.values())
            
            #   
            sorted_data = sorted(zip(words, counts), key=lambda x: x[1], reverse=True)
            sorted_words, sorted_counts = zip(*sorted_data)
            
            plt.bar(sorted_words, sorted_counts, color='salmon')
            plt.title('      ')
            plt.xlabel(' ')
            plt.ylabel('')
            plt.xticks(rotation=45, ha='right')
            plt.tight_layout()
            plt.show()
        else:
            print("    .")
